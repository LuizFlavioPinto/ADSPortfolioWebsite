Gerador De Senha Forte

Aluno: Luiz Flávio


import random
import string

def gerar_senha_forte(tamanho = random.randint(13,50)):
    if tamanho < 12:
        tamanho = 12
        print('É necessário no mínimo 12 caracteres para gerar uma senha forte.')

    letras_maiusculas = string.ascii_uppercase
    letras_minusculas = string.ascii_lowercase
    numeros = string.digits
    caracteres_especiais = string.punctuation

    senha = [
        random.choice(letras_maiusculas),
        random.choice(letras_minusculas),
        random.choice(numeros),
        random.choice(caracteres_especiais),
    ]

    todos_os_caracteres = letras_maiusculas + letras_minusculas + numeros + caracteres_especiais
    senha += random.choices(todos_os_caracteres, k=tamanho - 4)
    
    random.shuffle(senha)

    return ''.join(senha)


senha_gerada = gerar_senha_forte(13)
print(f"Senha gerada: {senha_gerada}")

print("-----CRIPTOGRAFIA DE MENSAGENS-----\n- Criptografa sua mensagem, transformando cada caractere no número de sua determinada posição no alfabeto.")
mensagem = input("\nDigite a mensagem que desejar criptografar (sem acentos, pontos ou números): ")
mensagem_cripto = [0] * len(mensagem)

i = 0

while i < len(mensagem):
   letra = mensagem[i]

   if 'A' <= letra <= 'Z':
       numero = ord(letra) - ord('A') + 1
       mensagem_cripto[i] = numero

   elif 'a' <= letra <= 'z':
       numero = ord(letra) - ord('a') + 1
       mensagem_cripto[i] = numero

   if letra == ' ':
       mensagem_cripto[i] = 0

   else:
       pass
   i += 1

print("Mensagem criptografada:", mensagem_cripto)

import math

# Variáveis para armazenar o estado da calculadora
historico = []
memoria = None

def menu_principal():
    print("\nCalculadora Simples")
    print("Escolha uma operação:")
    print("1. Soma")
    print("2. Subtração")
    print("3. Multiplicação")
    print("4. Divisão")
    print("5. Potência")
    print("6. Raiz quadrada")
    print("7. Armazenar na memória")
    print("8. Recuperar da memória")
    print("9. Limpar memória")
    print("0. Sair")

while True:
    menu_principal()
    escolha = input("Digite o número da operação (0 para sair): ")

    if escolha == '0':  # Sair
        print("Saindo...")
        break

    elif escolha in {'1', '2', '3', '4', '5', '6'}:  # Operações de cálculo
        while True:
            if escolha == '1':  # Soma
                num1 = float(input("Digite o primeiro número: "))
                num2 = float(input("Digite o segundo número: "))
                resultado = num1 + num2
                print(f"Resultado: {resultado}")
                historico.append(f"{num1} + {num2} = {resultado}")

            elif escolha == '2':  # Subtração
                num1 = float(input("Digite o primeiro número: "))
                num2 = float(input("Digite o segundo número: "))
                resultado = num1 - num2
                print(f"Resultado: {resultado}")
                historico.append(f"{num1} - {num2} = {resultado}")

            elif escolha == '3':  # Multiplicação
                num1 = float(input("Digite o primeiro número: "))
                num2 = float(input("Digite o segundo número: "))
                resultado = num1 * num2
                print(f"Resultado: {resultado}")
                historico.append(f"{num1} * {num2} = {resultado}")

            elif escolha == '4':  # Divisão
                num1 = float(input("Digite o primeiro número: "))
                num2 = float(input("Digite o segundo número: "))
                if num2 != 0:
                    resultado = num1 / num2
                    print(f"Resultado: {resultado}")
                    historico.append(f"{num1} / {num2} = {resultado}")
                else:
                    print("Erro: divisão por zero!")

            elif escolha == '5':  # Potência
                num1 = float(input("Digite a base: "))
                num2 = float(input("Digite o expoente: "))
                resultado = num1 ** num2
                print(f"Resultado: {resultado}")
                historico.append(f"{num1} ^ {num2} = {resultado}")

            elif escolha == '6':  # Raiz quadrada
                num = float(input("Digite um número para calcular a raiz quadrada: "))
                if num >= 0:
                    resultado = math.sqrt(num)
                    print(f"Resultado: {resultado}")
                    historico.append(f"Raiz quadrada de {num} = {resultado}")
                else:
                    print("Erro: não é possível calcular a raiz quadrada de um número negativo!")

            # Perguntar ao usuário se deseja continuar ou voltar ao menu
            continuar = input("\nDeseja continuar nessa operação? (s/n): ").lower()
            if continuar != 's':
                break

    elif escolha == '7':  # Armazenar na memória
        memoria = float(input("Digite um número para armazenar na memória: "))
        print(f"Valor armazenado na memória: {memoria}")

    elif escolha == '8':  # Recuperar da memória
        if memoria is not None:
            print(f"Valor na memória: {memoria}")
        else:
            print("Memória vazia!")

    elif escolha == '9':  # Limpar memória
        memoria = None
        print("Memória limpa!")

    else:  # Opção inválida
        print("Opção inválida!")

    # Exibir histórico de cálculos
    print("\nHistórico de cálculos:")
    for item in historico:
        print(item)

import re

def verificar_senha_forte(senha):
    if len(senha) < 8:
        return "A senha deve ter pelo menos 8 caracteres."
    if not re.search(r'[A-Z]', senha):
        return "A senha deve conter pelo menos uma letra maiúscula."
    if not re.search(r'[a-z]', senha):
        return "A senha deve conter pelo menos uma letra minúscula."
    if not re.search(r'[0-9]', senha):
        return "A senha deve conter pelo menos um número."
    if not re.search(r'[@#$%^&+=!]', senha):
        return "A senha deve conter pelo menos um caractere especial (@, #, $, %, ^, &, +, =, !)."
    return "Senha forte!"

senha = input("Digite a sua senha: ")
resultado = verificar_senha_forte(senha)
print(resultado)

# Entrada de dados
num = int(input("Digite um número inteiro: "))

while True:
    op = int(input("Qual operação você deseja realizar? Digite 0-(Para encerrar), 1-(Adição), 2-(Subtração), 3-(Multiplicação) ou 4-(Divisão): "))

    if op == 0:
        print(f"Programa encerrado!")
        break

#Verifica se o usuário digitou os números: 1, 2, 3 ou 4
    elif op in [1, 2, 3, 4]:   
        print(f"Tabuada do número: {num}.")

            if op == 1: # adição
                print(f"{num} + {i} = {num + i}")

        print("A quantidade total de notas deve ser maior que a quantidade de notas que você já possui.")
    else:
        entrada_notas = input(f"Digite as {quantidade_notas_existentes} notas separadas por espaço: ")
        entrada_pesos = input(f"Digite os {quantidade_total_notas} pesos correspondentes separados por espaço: ")

        notas = list(map(float, entrada_notas.split()))
        pesos = list(map(float, entrada_pesos.split()))

        if len(notas) != quantidade_notas_existentes or len(pesos) != quantidade_total_notas:
            print("Por favor, insira a quantidade correta de notas e pesos.")

            for peso in pesos:

    while True:
            idade = int(input(f"Digite a idade da pessoa {i + 1}: "))

                idades.append(idade)
                break
            else:
                print("Idade inválida. A idade deve ser um número inteiro positivo.")

total_idades = 0
maior_idade = idades[0]
menor_idade = idades[0]

    total_idades += idade
    
    # Verifica maior idade

        maior_idade = idade
        
    # Verifica menor idade

        menor_idade = idade

Calculadora Matriz 3X3

matriz = [[0 for _ in range (3)] for _ in range(3)]
submatriz1 = [[0 for _ in range (3)] for _ in range(3)]
submatriz2 = [[0 for _ in range (3)] for _ in range(3)]
produtoCadaDiagonalPrincipal = [0] * 3
produtoCadaDiagonalSecundaria = [0] * 3
somaCadaDiagonalPrincipal = 0
somaCadaDiagonalSecundaria = 0

produtoCadaDiagonalPrincipal[0] = 1
produtoCadaDiagonalPrincipal[1] = 1
produtoCadaDiagonalPrincipal[2] = 1
produtoCadaDiagonalSecundaria[0] = 1
produtoCadaDiagonalSecundaria[1] = 1
produtoCadaDiagonalSecundaria[2] = 1

for i in range(0,3):
    for j in range(0,3):
        matriz[i][j] = int(input(f"Digite um número que será o elemento [{i+1}][{j+1}] da matriz 3x3 para calcular seu determinante: "))

print("\n")

for i in range(0,3):
    for j in range(0,3):
        print(matriz[i][j], end="\t")
    print()

for i in range(0,3):
    submatriz1[i][j] = matriz[i][j]
    submatriz1[i][0] = matriz[i][1]
    submatriz1[i][1] = matriz[i][2]
    submatriz1[i][2] = matriz[i][0]

    submatriz2[i][j] = matriz[i][j]
    submatriz2[i][0] = matriz[i][2]
    submatriz2[i][1] = matriz[i][0]
    submatriz2[i][2] = matriz[i][1]

for i in range(0,3):
    for j in range(0,3):
        if(i == j):
            produtoCadaDiagonalPrincipal[0] *= matriz[i][j]
            produtoCadaDiagonalPrincipal[1] *= submatriz1[i][j]
            produtoCadaDiagonalPrincipal[2] *= submatriz2[i][j]
    
for i in range(0,3):
    somaCadaDiagonalPrincipal += produtoCadaDiagonalPrincipal[i]
    
for i in range(0,3):
    for j in range(0,3):
        if(i + j == 2):
            produtoCadaDiagonalSecundaria[0] *= matriz[i][j]
            produtoCadaDiagonalSecundaria[1] *= submatriz1[i][j]
            produtoCadaDiagonalSecundaria[2] *= submatriz2[i][j]
    
for i in range(0,3):
    somaCadaDiagonalSecundaria += produtoCadaDiagonalSecundaria[i]
    
determinante = somaCadaDiagonalPrincipal - somaCadaDiagonalSecundaria

print(f"\nDeterminante da matriz 3x3: {determinante}")
